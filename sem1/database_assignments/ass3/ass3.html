Assignment 3

(A)Without using primary key (Without Constraints)

Step 1 : Create A database

create database [database name];

For Example

create database kmc;

OR

Use the database which is already available by using this command

show databases;

To use Available database type this command

use databasename;

like

use kmc;

Step 2 : Create table in database

create table employee(eno int,ename varchar(20),city varchar(20),salary varchar(20));

Step 3 : To insert values into database

insert into employee values(103,"shreyas","khopoli","100000");

Step 4 : To see All employee records

select * from employee;

Step 5 : Select the city of employees

select city from employee;

Step 6 : Select perticular employee using where clause

select * from employee where eno=103;

Step 7 : To update particular employee

update employee set salary=200000 where eno=103;

Step 8 : To delete particular employee

delete from employee where eno=103;
(B) With using Primary Key (With Constraints)

Step 1 : Creating table using primary key

create table department(dno int primary key,dname varchar(20));

Step 2 : Creating table using foreign key

create table employees(eno int primary key,ename varchar(20),city varchar(20),dno int,FOREIGN KEY(dno)REFERENCES dept(dno));

Step 3 : Inserting value in department table

insert into department values(1,'IT');

Step 4 : Viewing all records from departments

select * from department;

Step 5 : Inserting value in employees table

insert into employess values(101,'shreyas','khopoli',1);

Step 6 : Viewing all records from employees

select * from employess;
